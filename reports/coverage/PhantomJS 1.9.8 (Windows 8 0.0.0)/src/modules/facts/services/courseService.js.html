define([], function() {
	'use strict';
	return ['Restangular', function(Restangular) {
		var service = this;
		
		service.currentCourse = {
			courseId: -1,
			course: {},
			loaded: false
		};
		
		service.getCourse = function(courseId) {
			if(!service.currentCourse.loaded || service.currentCourse.courseId !== courseId) {
				// Return the restangularized version
				return Restangular.one('courses', courseId).get().$object;
			} // Return the cached page version.
			return service.currentCourse.course;
		};
		
		service.loadCourse = function(courseId) {
			return Restangular.one('courses', courseId).get().then(function(_course) {
				service.currentCourse.loaded = true;
				service.currentCourse.courseId = courseId;
				service.currentCourse.course = _course;
 
				return service.currentCourse.course;
			});
		};
		
		return service;
	}];
});
