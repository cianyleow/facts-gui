define([], function() {
	'use strict';
	return [function() {
		var service = this;
		
		var colorList = ["#F44336", "#2196F3", "#4CAF50", "#FFEB3B", "#FF9800", "#795548", "#9E9E9E"];
		service.getPieChartLines = function(dataList) {
			var lines = [];
			angular.forEach(dataList, function(data, index) {
				var line = {label: data.name, value: data.maxMark, color: colorList[index % colorList.length]};
				lines.push(line);
			});
			return lines;
		};
		return service;
	}];
});
